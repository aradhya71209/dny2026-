  function typingEffect(text, element) {
    element.innerHTML = "";
    let i = 0;
    function type() {
      if (i < text.length) {
        element.innerHTML += text[i] === '\n' ? '<br>' : text[i];
        i++;
        setTimeout(type, 100);
      }
    }
    type();
  }
